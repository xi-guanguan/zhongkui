## 坐标系统

```lua
local CELL = 0.5          -- 每格半米
local GRID_W, GRID_H = 40, 40  -- 40×40 格
local HALF_W = GRID_W * CELL * 0.5
local HALF_H = GRID_H * CELL * 0.5

--- 网格坐标 → 世界坐标（格中心）
function GridToWorld(gx, gz)
    local wx = (gx - 0.5) * CELL - HALF_W
    local wz = (gz - 0.5) * CELL - HALF_H
    return wx, wz
end

--- 世界坐标 → 网格坐标（四舍五入）
function WorldToGrid(wx, wz)
    local gx = math.floor((wx + HALF_W) / CELL + 0.5) + 1
    local gz = math.floor((wz + HALF_H) / CELL + 0.5) + 1
    return gx, gz
end
```

## 占据表

```lua
local occupied = {}  -- occupied["gx,gz"] = buildingId

function IsOccupied(gx, gz)
    return occupied[gx .. "," .. gz] ~= nil
end

function SetOccupied(gx, gz, id)
    occupied[gx .. "," .. gz] = id
end

function ClearOccupied(gx, gz)
    occupied[gx .. "," .. gz] = nil
end
```

## 旋转与多格中心

```lua
--- 计算多格建筑中心偏移（旋转后）
--- rotation: 0/90/180/270
function GetPlacementCenter(def, gx, gz, rotation)
    local sx, sz = def.sizeX, def.sizeZ
    if rotation == 90 or rotation == 270 then
        sx, sz = sz, sx  -- 交换
    end
    local cx = gx + (sx - 1) * 0.5
    local cz = gz + (sz - 1) * 0.5
    return cx, cz
end

--- 获取多格建筑占据的所有格子
function GetOccupiedCells(def, gx, gz, rotation)
    local sx, sz = def.sizeX, def.sizeZ
    if rotation == 90 or rotation == 270 then sx, sz = sz, sx end
    local cells = {}
    for dx = 0, sx - 1 do
        for dz = 0, sz - 1 do
            table.insert(cells, { gx + dx, gz + dz })
        end
    end
    return cells
end
```

## 模型创建（三种方式）

```lua
--- 方式1: 导入的 mdl 模型
function CreateFromMdl(parentNode, def, wx, wz, rotation)
    local node = parentNode:CreateChild("Building")
    node.position = Vector3(wx, def.height * 0.5, wz)
    node.rotation = Quaternion(rotation, Vector3.UP)
    local model = node:CreateComponent("StaticModel")
    model.model = cache:GetResource("Model", def.model)
    model.material = cache:GetResource("Material", "Materials/Building.xml")
    return node
end

--- 方式2: Box.mdl 缩放
function CreateFromBox(parentNode, def, wx, wz, rotation)
    local node = parentNode:CreateChild("Building")
    node.position = Vector3(wx, def.height * 0.5, wz)
    node.rotation = Quaternion(rotation, Vector3.UP)
    node.scale = Vector3(def.sizeX * CELL, def.height, def.sizeZ * CELL)
    local model = node:CreateComponent("StaticModel")
    model.model = cache:GetResource("Model", "Models/Box.mdl")
    return node
end

--- 方式3: Prefab XML
function CreateFromPrefab(parentNode, prefabPath, wx, wz, rotation)
    local file = cache:GetResource("XMLFile", prefabPath)
    local node = parentNode:CreateChild("")
    node:LoadXML(file:GetRoot())
    node.position = Vector3(wx, 0, wz)
    node.rotation = Quaternion(rotation, Vector3.UP)
    return node
end
```

## Place / Remove

```lua
local buildings = {}  -- buildings[id] = { node, def, gx, gz, rotation }
local nextId = 1

function Place(defName, gx, gz, rotation)
    local def = Registry[defName]
    if not def then return nil, "unknown_type" end

    rotation = rotation or 0
    local cells = GetOccupiedCells(def, gx, gz, rotation)

    -- 检查所有格子是否可用
    for _, cell in ipairs(cells) do
        if IsOccupied(cell[1], cell[2]) then
            return nil, "occupied"
        end
    end

    -- 计算世界坐标
    local cx, cz = GetPlacementCenter(def, gx, gz, rotation)
    local wx, wz = GridToWorld(cx, cz)

    -- 创建模型
    local node = CreateFromMdl(scene_, def, wx, wz, rotation)

    -- 标记占据
    local id = nextId; nextId = nextId + 1
    for _, cell in ipairs(cells) do
        SetOccupied(cell[1], cell[2], id)
    end

    buildings[id] = { node = node, def = def, gx = gx, gz = gz, rotation = rotation, cells = cells }
    return id
end

function Remove(id)
    local b = buildings[id]
    if not b then return end
    for _, cell in ipairs(b.cells) do
        ClearOccupied(cell[1], cell[2])
    end
    b.node:Remove()
    buildings[id] = nil
end
```

```lua
local ROAD_VARIANTS = {
    straight_ns = "Models/Road_Straight.mdl",    -- 南北直道

    corner      = "Models/Road_Corner.mdl",
    t_junction  = "Models/Road_T.mdl",
    crossroad   = "Models/Road_Cross.mdl",
    dead_end    = "Models/Road_DeadEnd.mdl",
}

--- 检测相邻道路
function GetRoadNeighbors(gx, gz)
    local n = IsRoad(gx, gz - 1)  -- 北
    local s = IsRoad(gx, gz + 1)  -- 南
    local e = IsRoad(gx + 1, gz)  -- 东
    local w = IsRoad(gx - 1, gz)  -- 西
    return n, s, e, w
end

--- 根据邻居选择变体+旋转
function SelectRoadVariant(n, s, e, w)
    local count = (n and 1 or 0) + (s and 1 or 0) + (e and 1 or 0) + (w and 1 or 0)
    if count == 4 then return "crossroad", 0
    elseif count == 3 then
        -- T 型路口（缺口方向决定旋转）
        if not n then return "t_junction", 180
        elseif not s then return "t_junction", 0
        elseif not e then return "t_junction", 270
        else return "t_junction", 90 end
    elseif count == 2 then
        if n and s then return "straight_ns", 0
        elseif e and w then return "straight_ew", 90
        else -- 拐角（4种）
            if n and e then return "corner", 0
            elseif e and s then return "corner", 90
            elseif s and w then return "corner", 180
            else return "corner", 270 end
        end
    elseif count == 1 then return "dead_end", (n and 0 or s and 180 or e and 90 or 270)
    else return "straight_ns", 0 end  -- 孤岛默认
```

## 批量布局与查询

```lua
--- 批量放置（从 layout table）
function BatchPlace(layout)
    for _, entry in ipairs(layout) do
        Place(entry.type, entry.gx, entry.gz, entry.rotation)
    end
end

--- 查询某格子上的建筑
function QueryCell(gx, gz)
    local id = occupied[gx .. "," .. gz]
    return id and buildings[id] or nil
end
```

## 序列化与存档

```lua
--- 导出所有建筑为可序列化 table
function Export()
    local data = {}
    for id, b in pairs(buildings) do
        table.insert(data, {
            id       = id,
            type     = b.def.name or "unknown",
            gx       = b.gx,
            gz       = b.gz,
            rotation = b.rotation,
        })
    end
    return data
end

--- 从存档恢复
function Import(data)
    for _, entry in ipairs(data) do
        Place(entry.type, entry.gx, entry.gz, entry.rotation)
    end
end
```
